Converts the from pointer to a NonNull.

§Example

use abi_stable::pointer_trait::immutable_ref;

use std::ptr::NonNull;

const X: NonNull<i8> = immutable_ref::to_nonnull(&3i8);
unsafe {
    assert_eq!(*X.as_ref(), 3i8);
}
