Macro ros_nalgebra::ros_nalgebra

macro_rules! ros_nalgebra {
    ($($ns:ident)::+) => { ... };
}
Expand description

Generate Point/Vector3/Quaternion/Pose/Transform converter in mod $ns. $ns is the namespace for the rust messages. For example msg.

§Example

mod msg {
    rosrust::rosmsg_include!(
        geometry_msgs/Point,
        geometry_msgs/Pose,
        geometry_msgs/Quaternion,
        geometry_msgs/Transform,
        geometry_msgs/Vector3,
    );
}
ros_nalgebra::ros_nalgebra!(msg);
